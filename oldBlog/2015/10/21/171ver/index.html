JavaScript：编程练习-动态的添加元素+更改颜色

学习代码以及文章来自http://www.imooc.com/code/1636

编程练习



制作一个表格，显示班级的学生信息。


要求：


1. 鼠标移到不同行上时背景色改为色值为 #f2f2f2，移开鼠标时则恢复为原背景色 #fff


2. 点击添加按钮，能动态在最后添加一行


3. 点击删除按钮，则删除当前行


 


学习经验


基本上没太大的难度 唯一注意的就是 color函数


每次创建了一个新的元素都需要调用color函数


color函数里的鼠标移入事件和鼠标移出事件可以使用this来指向 也可以使用闭包来指向


闭包指向会出现问题..还是用this方便简单


 


 


 


 


<!DOCTYPE html>
<html>
 <head>
  <title> new document </title>
  <meta http-equiv=”Content-Type” content=”text/html; charset=utf-8”/>
  <script type=”text/javascript”>

      window.onload = function(){
            color()
     // 鼠标移动改变背景,可以通过给每行绑定鼠标移上事件和鼠标移除事件来改变所在行背景色。




     }

      // 编写一个函数，供添加按钮调用，动态在表格的最后一行添加子节点；

    function color(){
        var get_tr=document.getElementsByTagName(“tr”)
        for(var i=0;i<get_tr.length;i++){
            get_tr[i].onmouseover=function(num){

                this.style.background=”#f2f2f2”}

            get_tr[i].onmouseout=function(num1){


                this.style.background=”#fff”

            }
        }
    }
       function createB(){
           var get_table=document.getElementById(“table”)//获取id
           // var get_table_tr=get_table.getElementsByTagName(“tr”)
           var last_tr=get_table.lastChild//获取到table最后一个子节点
           var new_tr=document.createElement(“tr”)
           var new_td_xh=document.createElement(“td”)
           var new_td_xm=document.createElement(“td”)
           var new_td_cz=document.createElement(“td”)
           var new_td_a=document.createElement(“a”)
           console.log(last_tr);

           var save_a_href=”javascript:;”//保存a的href属性
           var save_a_onclick=”delElem(this)”//保存a的onclick属性
           var save_a_into=”删除”//保存a的内容
           //以上三步可以不这样写
           //方式有很多种
           var new_xuehao=prompt(“please input xuehao”)//创建一个弹出输入框
           var new_xingmin=prompt(“please input name”)
           last_tr.appendChild(new_tr)//添加tr元素
           new_tr.appendChild(new_td_xh)
           new_tr.appendChild(new_td_xm)
           new_tr.appendChild(new_td_cz)
           new_td_cz.appendChild(new_td_a)



           new_td_xh.innerHTML=new_xuehao//学号的innerHTML等于弹出框的内容

           new_td_a.href=save_a_href//设置a标签的href属性
           new_td_a.setAttribute(“onclick”, save_a_onclick)//设置a标签的onclick事件
           new_td_a.innerHTML=save_a_into//设置a的内容
           color()//再次调用color函数…






       }

        var parent_tr=elem.parentNode.parentNode//获取到tr节点

        var parent_table=parent_tr.parentNode
        parent_table.removeChild(parent_tr)//在table节点里删除tr节点
        console.log(parent_table)

    }
