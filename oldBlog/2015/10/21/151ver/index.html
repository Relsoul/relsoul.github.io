块语句与try catch


//块语句
    //常用句组合0~多个语句,用花括号定义
    //无块级作用域
    // {a:1,b:2}//error
    var o={a:1,b:2}//ok
    //

    function test(){
        var a=b=1;
        //实际上隐性的创建了一个全局变量b
    }
    test();
    console.log(typeof a);//undefined
    console.log(typeof b)//number

    try{
        throw ‘test’;

    }catch(ex){
        console.log(ex)//test
    }finally{
        console.log(‘finally’)
    }

    try{
        //do sth

    }finally{
        console.log(‘finally’)
        //不管中间有没有异常 都要执行finally
    }

    try{
        try{
            throw new Error(‘oops’);
        }finally{console.log(‘finally’);}
    }
    catch(ex){console.error(‘outer’,ex.message)}
     //执行结果 先finally 然后outer 最后oops
     //内部catch抛出异常了给外部处理 会先执行内部的finally
