JS之探测器实例 

!function(global){
                function DetectorBase(configs){
                    //如果this不是DetectorBase本身的话也就是 this为DetectorBase prototype
                    //必须使用new DetectorBase 阻止外部直接调用
                    if(!this instanceof DetectorBase){
                        throw new Error(“Do not invoke without new|无法调用新detectorBase”)
                    }
                    this.configs=configs;
                    this.analyze();//创建的时候是一个实例 实例.analyze() 解析一下数据
                }
                DetectorBase.prototype.detect=function(){
                    throw new Error(“Not implemented|未执行”)
                }
                DetectorBase.prototype.analyze=function(){
                    console.log(“analyzing..”);
                    this.data=”###date###”
                }
                function LinkDetector(links){
                    if(!this instanceof LinkDetector){
                        throw new Error(“Do not invoke without new|无法调用新detectorBase”)
                    }
                    this.links=links;
                    DetectorBase.apply(this,arguments)
                }
                function ContainerDetector(containers){
                    if(!this instanceof ContainerDetector){
                        throw new Error(“Do not invoke without new|无法调用新detectorBase”)
                    }
                    this.containers=containers
                    DetectorBase.apply(this,arguments);
                }
                //inherit first
                //实际上就是Object.create()
                //先实现原型链继承 再实现link,containerDetector属性
                inherit(LinkDetector,DetectorBase);
                inherit(ContainerDetector,DetectorBase)

    //expand later 扩展
    LinkDetector.prototype.detect=function(){
        console.log("Loading data:"+this.data)
        console.log("Link detection started.")
        console.log("Scaning links:"+this.links)
    }
    ContainerDetector.prototype.detect=function(){
        console.log("Loading data:"+this.data)
        console.log("Container detection started")
        console.log("Scanning containers:"+this.containers)
    }

    //prevent from being altered 防止被修改
    Object.freeze(DetectorBase)
    Object.freeze(DetectorBase.prototype)
    Object.freeze(LinkDetector)
    Object.freeze(LinkDetector.prototype)
    Object.freeze(ContainerDetector)
    Object.freeze(ContainerDetector.prototype)

    //export to global object 导出到全局对象
    //暴露到外面 同时不可改写
    Object.defineProperties(global,{
        LinkDetector:{value:LinkDetector},
        ContainerDetector:{value:ContainerDetector},
        DetectorBase:{value:DetectorBase}
    })
    function inherit(subClass,superClass){
        subClass.prototype=Object.create(superClass.prototype)
        subClass.prototype.constructor=subClass
    }

先聊聊JS中的正则


JS中的正则有两种风格





  1. var jsfg=new RegExp(“a”,“gi”) //js风格

  2. var perlfg=/a/gi //perlfg风格





JS风格是通过一个正则构造器来构造的 new RegExp(“正则”,”是否区分大小写，全局搜索等等~”)


perlfg风格通过 /正则/是否区分大小写，全局搜索等等~ 这样的形式


okay开始我们第一个正则:不区分大小写全局替换字符串中的a





  1. var str=“abcAbcAABBCC”

  2. var re=/a/gi

  3. var rep=str.replace(re,“替换”)

  4. console.log(rep) //替换bc替换bc替换替换BBCC




 


上面的无论是A还是a全部被替换成了一个字符串“替换”


我们主要注意/a/gi后的gi


g:全局匹配 不加g则匹配一次就不匹配了


i:不区分大小写 不加就只匹配小写a


以上你就完成了第一个正则匹配 是不是很激动？ 别急 咋们继续


 


范例二:查找字符串第一个出现的数字





  1. var str4=‘adsf 23 23 wcs’

  2. var re2=/\d/

  3. console.log(str4.search(re2))//5




okay 火星文出现了 \d 是什么意思？ 先放着咋们先了解下基础


 


先说说正则里面的圆括号，方括号





  1. var str8=“apc aapc ccpc dbpc”

  2. var re6=/[abc]pc/g //a或b或c apc bpc

  3. var re7=/apc|bpc|cpc/g

  4. console.log(str8.match(re6))//apc apc cpc bpc

  5. console.log(str8.match(re7)) //apc apc cpc bpc





match这个函数的功能是以数组的形式返回匹配到的元素.


/apc|bpc|cpc/g 中”|”是什么意思？


我们知道js中用||代表或 那么在js中用单杠代表或


这里的/[abc]pc/g 啥意思呢？


“g”我们知道是全局匹配


/[abc]pc/是shenmegui？


[abc]pc的意思是 a或b或c开头的英文字母后面接一个pc 比如apc,bpc,cpc;可以匹配到这几种


可以得出方括号里的内容都是或的 比如我们要匹配0-9a-z下划线那么可以这样写





  1. var re7=/[0-9a-z]/




而上面这段看起来有点长的正则可以简写为 \w





  1. var re7=/\w/ //等同于上面的正则




再来说说 我们只匹配除0-9以外的东西应该怎么写？





  1. var re8=/[^0-9]/




注意我在方括号的开头加了一个^符号 而这段正则可以这样写





  1. var re8=/\D/ //注意我这里是大写的\D 不是小写的\d




大写小写肯定是有区别的 他们之间区别也很简单


我们知道\D是除0-9以外的东西 那么小写的\d就是 匹配0-9 也就是 /[0-9]/


同理 \w是匹配0-9a-z下划线 那么\W 就是匹配0-9a-z下划线除外的东西 也就是/[^0-9a-z]/

来点复杂的范例4:过滤HTML标签





  1. var str10=“<p>dasdsa</p>nice <br> test</br>”

okay 先别急着看 我们先了解正则火星文中的量词.说白了就是出现次数



量词:个数

{n}正好出现n次

[1-9]\d{8}只出现八次的数字 第一位为1-9数字

{n,m}最少n次最多m次

[1-9]\d{4,10}

{n,}最少n次，最多不限

+{1,}

?{0,1}最少0次最多一次

*{0,}可以有 最多可以有无限 0次也算 匹配不推荐用


okay 从上表知道{n,m}代表量词 那么上面的范例其实也可以这样改造





  1. var re13=/<[^<>]+>/g

  2. var re13=/<[^<>]{1,}>/g





okay现在知道+号是什么意思了 现在来分析下这个正则





  1. var re13=/<[^<>]+>/g

  2. <[^<>]+>/g 匹配尖括号<[里除尖括号以外的所有字符]字符出现次数最少为一次>





如果不加那个+号会变成什么情况呢？ 这个问题留给大家思考一下


okay 学到这里正则基本上也就入门了. 现在回过头来看看文章开头提到的邮箱匹配正则





  1. /(^\w+)@([a-z0-9]+).([a-z]+$)/gi






这里的()起一个分组的作用 你可以看成数学中的加减乘除所以上例的正则去掉圆括号意思其实是一样的 先分析@前面的正则





  1. (^\w+) //=^\w+





\w是啥意思？匹配任何字类字符，包括下划线。与“[A-Za-z0-9]”等效。 所以邮箱的@前可以有以上的字符并且出现的次数都不限定最少出现一次 但是我为什么要在正则前面加一个^ 这个问题放在后面再说 咋们继续






  1. ([a-z0-9]+). //=[a-z0-9]+.





这里用到了一个.意思为转义字符 我们知道正则中.是匹配任意字符，我们只需要匹配一个单纯的”.”号所以需要用到一个转义字符\ 当然匹配[]也是需要用到转义字符的 [ ]

@gtb123.xxx 这里的gtb123则是匹配到的内容 我们知道邮箱@号后面可以跟数字和字符所以我们要写上可以匹配到数字和字符


现在来看看.后面的正则





  1. ([a-z]+$) //=[a-z]+$





我们知道.后面只能跟英文 比如 .com .cn .net

这里的$是啥意思？ 这里的$代表字符串的结尾 所以这个正则的意思就是 字符串的结尾必须 com net cn 英文结尾的


现在回过头来看看@开头的内容





  1. (^\w+) //=^\w+





这里的^不是代表否,非的意思了 在方括号里面[^0-9]的^代表否，非 那么没在方括号里面的代表的就是字符串开头

所以这个正则的意思就是 字符串必须是以下划线,数字,字母开头的


结合起来这串正则的意思就





  1. /(^\w+)@([a-z0-9]+).([a-z]+$)/gi

  2. 下划线,数字,字母@字母,数字.英文

  3. 比如

  4. test1213_123@AbDd.cn





有了这个正则那么我们应该怎么用来判断呢？ js里有一个test方法





  1. var str10=“sd_blue@etst.com”;

  2. var re13=/(^\w+)@([a-z0-9]+).([a-z]+$)/gi

  3. console.log(re13.test(str10))// true






js中的test方法 如果满足要求则符合true 注意就算字符串中的一部分符合要求也会返回true的.

圆括号不仅仅有分组的意思 还涉及到一个捕获,我们知道match会返回一个数组 举个简单例子





  1. var str19=“2030”;

  2. var re19=/\d+[+-\\/]\d+/

  3. var mat1=str19.match(re19)

  4. console.log(mat1) //[“2030”, index: 0, input: “2030”]




上面是一串匹配数字,符号,数字的正则 看起来比较奇怪 但是还是能够匹配到的. 返回的是一个数组 注意！这时我们加上一个圆括号来分组下





  1. var str19=“2030”;

  2. var re19=/(\d+)([+-\\/])(\d+)/

  3. var mat1=str19.match(re19)

  4. console.log(mat1) //[“2030”, “20”, ““, “30”, index: 0, input: “2030”]




这里返回的数组是[“2030”, “20”, ““, “30”, index: 0, input: “2030”]


0:”2030”


1:”20”


2:”“


3:”30”


可以得出数组[0]为匹配元素 从1开始就得出了分组捕获的内容


上面的正则第一个圆括号分组捕获的是符号前面的数字 第二个圆括号捕获的是符号 第三个圆括号捕获的是符号后面的数字


捕获组还能反向引用，即在表达式中直接使用某个分组的内容。





  1. var str20=“20-20”;

  2. var re20=/(\d+)([+-*\/])(\1)/

  3. var mat2=str20.match(re20)

  4. console.log(mat2) //[“20-20”, “20”, “-“, “20”, index: 0, input: “20-20”]




这里第三个分组 也就是(\1)引用的是第一个分组(\d+)捕获到的内容 同理 如果要引用分组2捕获到的内容那么





  1. var str20=“20-20”;

  2. var re20=/(\d+)([+-*\/])(\2)/

  3. var mat2=str20.match(re20)

  4. console.log(mat2) //[“20-20”, “20”, “-“, “20”, index: 0, input: “20-20”]




改为\2即可 同理 引用3,4,5,… \n即可 别忘记了分组捕获是从1开始的 0为匹配内容


置换前、后两个数的位置





  1. var str21=“178-334”;

  2. var re21=/(\d+)([+-*\/])(\d+)/

  3. var rep3=str21.replace(re21,“$3$2$1”)

  4. console.log(rep3)//334-178




在replace函数中用$1,2,3,n代表分组

折线图的一些说明

tooltip

提示框，鼠标悬浮交互时的信息提示，比如

{bollean}show true|false 显示|隐藏
{number}showDelay 20 显示延迟

其他的信息请看官网范例

legend

图例，每个图表最多仅有一个图例 官网范例

toolbox

工具箱，每个图表最多仅有一个工具箱 官网范例

calculable

拖拽计算

{boolean} false，是否启用拖拽重计算特性，默认关闭

xAis/yAxis

点击标题参考文档

series

驱动图表生成的数据内容数组，数组中每一项为一个系列的选项及数据，其中个别选项仅在部分图表类型中有效，请注意适用类型 点击标题直接去官网看文档吧
