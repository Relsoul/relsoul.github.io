!function(global){
                function DetectorBase(configs){
                    //如果this不是DetectorBase本身的话也就是 this为DetectorBase prototype
                    //必须使用new DetectorBase 阻止外部直接调用
                    if(!this instanceof DetectorBase){
                        throw new Error(“Do not invoke without new|无法调用新detectorBase”)
                    }
                    this.configs=configs;
                    this.analyze();//创建的时候是一个实例 实例.analyze() 解析一下数据
                }
                DetectorBase.prototype.detect=function(){
                    throw new Error(“Not implemented|未执行”)
                }
                DetectorBase.prototype.analyze=function(){
                    console.log(“analyzing..”);
                    this.data=”###date###”
                }
                function LinkDetector(links){
                    if(!this instanceof LinkDetector){
                        throw new Error(“Do not invoke without new|无法调用新detectorBase”)
                    }
                    this.links=links;
                    DetectorBase.apply(this,arguments)
                }
                function ContainerDetector(containers){
                    if(!this instanceof ContainerDetector){
                        throw new Error(“Do not invoke without new|无法调用新detectorBase”)
                    }
                    this.containers=containers
                    DetectorBase.apply(this,arguments);
                }
                //inherit first
                //实际上就是Object.create()
                //先实现原型链继承 再实现link,containerDetector属性
                inherit(LinkDetector,DetectorBase);
                inherit(ContainerDetector,DetectorBase)

    //expand later 扩展
    LinkDetector.prototype.detect=function(){
        console.log("Loading data:"+this.data)
        console.log("Link detection started.")
        console.log("Scaning links:"+this.links)
    }
    ContainerDetector.prototype.detect=function(){
        console.log("Loading data:"+this.data)
        console.log("Container detection started")
        console.log("Scanning containers:"+this.containers)
    }

    //prevent from being altered 防止被修改
    Object.freeze(DetectorBase)
    Object.freeze(DetectorBase.prototype)
    Object.freeze(LinkDetector)
    Object.freeze(LinkDetector.prototype)
    Object.freeze(ContainerDetector)
    Object.freeze(ContainerDetector.prototype)

    //export to global object 导出到全局对象
    //暴露到外面 同时不可改写
    Object.defineProperties(global,{
        LinkDetector:{value:LinkDetector},
        ContainerDetector:{value:ContainerDetector},
        DetectorBase:{value:DetectorBase}
    })
    function inherit(subClass,superClass){
        subClass.prototype=Object.create(superClass.prototype)
        subClass.prototype.constructor=subClass
    }
}(this)

DOM编程艺术之数据通信-HTTP协议 

一:HTTP
            -客户端向服务器发送包含命令和URL的HTTP请求报文
            -服务器返回一个包含事务结果的HTTP响应报文

二:请求报文格式
    -头行 
        -GET music.163.com HTTP/1.1
          方式  主机地址        版本
    -头部
        -许多键值对构成
            -Accept:浏览器端接受的类型
            -Accept-Encoding:浏览器端编码类型
            -Accept-Language:浏览器端可以接受的语言类型
            -Cache-Control:浏览器是否可以缓存
            -Connection:是否持续链接
            -Cookie:浏览器向服务器端发送的cookie
            -Host:主机地址
            -Pragma:
            -User-Agent:浏览器版本

三:响应报文格式
    -头行
        -HTTP/1.1  200    OK
         HTTP版本  状态码    HTTP状态码描述
    -头部
        -Expires:缓存有效时间
        -Server:服务器端HTTP服务器
    -主体
        -完整的XML页面

四:常用HTTP方法
    -GET
        -从服务器获取一份稳定 不包含主体
    -POST
        -向服务器发送需要处理的数据 包含主体
    -PUT
        -将请求的主体部分储存在服务器上 包含主体
    -DELETE
        -从服务器上删除一份稳定 不包含主体
    -HEAD
        -只从服务器获取文档的首部 不包含主体
    -TRACE
        -对可能经过代理服务器传送到服务器上的报文进行追踪 不包含主体
    -OPTIONS
        -决定可以在服务器上执行哪些方法 不包含主体

五:URL构成
    -http://www.163.com:8080/index.html?=admin&amp;lang=zh-CN#news
    -protocol
        -http
    -host
        -www.163.com:8080
    -pathname
        -index.html
    -hostname
        -www.163.com
    -port
        -8080
    -search
        -?r=admin&amp;lang=zh-CN
    -hash
        -news

六:HTTP版本
    -HTTP/0.9
        -1991年 HTTP原型，有很多设计缺陷
    -HTTP/1.0
        -很快取代HTTP/0.9，成为第一个广泛应用的版本
    -HTTP/1.0+
        -添加持久的keep-alive连接，虚拟主机支持，以及代理连接支持，成为非官方的事实标准
    -HTTP/1.1
        -校正了结构性缺点，明确语义，引入重要的性能优化措施，删除不好特性，是当前使用的版

七:常见HTTP状态码
    -200
        -请求成功。一般用于GET和POST方法 原因短语:OK
    -301
        -资源移动，所请求资源自动到新的URL，浏览器自动跳转到新的URL moved Permanently
    -304
        -未修改。所请求资源未修改，浏览器读取缓存数据 Not Modified
    -400
        -请求语法错误，服务器无法理解 Bad Request
    -404
        -未找到资源，可以设置个性404页面 Not Found
    -500
        -内部服务器错误 Internal Server Error

            if(!data)return "";
            var pairs=[];
            for (var name in data) {
                if(!data.hasOwnProperty(name))continue;
                if(typeof data[name]==='function') continue;
                var value=data[name].toString();
                name=encodeURIComponent(name);
                value=encodeURIComponent(value);
                pairs.push(name+"="+value);
            }
            return pairs.join("&amp;")

折线图的一些说明

tooltip

提示框，鼠标悬浮交互时的信息提示，比如

{bollean}show true|false 显示|隐藏
{number}showDelay 20 显示延迟

其他的信息请看官网范例

legend

图例，每个图表最多仅有一个图例 官网范例

toolbox

工具箱，每个图表最多仅有一个工具箱 官网范例

calculable

拖拽计算

{boolean} false，是否启用拖拽重计算特性，默认关闭

xAis/yAxis

点击标题参考文档

series

驱动图表生成的数据内容数组，数组中每一项为一个系列的选项及数据，其中个别选项仅在部分图表类型中有效，请注意适用类型 点击标题直接去官网看文档吧

                    -返回id或name为指定名称的表单控件(除图片按钮input type="image")

                    -如果有多个同名元素，则返回这些元素的动态节点集合
                    -一旦用指定名称取过该元素，则不管该元素的id或者name怎么变化，只要节点还在页面上均可以用原名获取该元素
                    -非动态集合所以可以取到已经设置过的元素如
                        -testForm["a"]=&gt;name="b" testForm.elements["a"]=&gt;null

                        -可重置元素:input,keygen,output,select,textarea
