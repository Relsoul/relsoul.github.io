JS程序设计之面向对象原型链 

面向对象原型与原型链

一:思考
    -function Teacher(){
        this.courses=[];
        this.job='teacher';
        this.setName=function(name){
            this.name=name
        }
        this.addCourse=function(course){
            this.courses.push(course)
        }
    }

    var bill=new Teacher();
    bill.setName('Bill');
    bill.addCourse('Math');

    var tom=new Teacher();
    tom.setName('Tom');
    tom.addCourse('physics')

    //以上tom和bill的公共属性都耗用了内存 因为每次都需要生成每个实例的公共方法

二:原型
    -prototype
    -function Teacher(){

        this.courses=[];
    }
    Teacher.prototype={
        job:'teacher',
        setName:function(name){
            this.name=name;
        }
        addCourse:function(course){
            this.courses.push(course)
        }
    }
    //生成的实例有一个_proto_执行构造器的原型

三:原型链
    -实例的_proto_都指向teacher.prototype,teacher._proto_指向Object.prototype
    -Object.prototype指向null

四:原型链-属性查找
    -从原型链从下往上查找

五:原型链-属性修改
    -直接修改对象属性，永远修改对象本身属性，无论是来源于原型还是本身，如本身无这个属性则增加这个属性至本身
    -tom.job="soul" //tom本身无job属性,在tom对象上创建一个job属性同时设置job为soul
    -Teacher.prototype.job="soul"//修改原型对象上的属性
    -bill.job//soul

六:原型链-属性删除
    -delete tom.job//删除了自身job属性 返回的是原型上的job属性
    -delete tom.job//无法删除到原型上的属性 不会有任何变化

七:思考
    -如何判断一个属性来自于对象本身？
        -tom.hasOwnProperty("job")//来源于对象本身返回true 反之返回false

八:ES5中的原型继承
    -Object.create(proto[,propertiesObject])//第二个参数为创建的实例对象设置属性方法
    -var soul={
        //私有属性与方法
        job:"teacher",
        courses:[],
        addCourse:function(course){
            this.course.push(course)
        }
    }
    var bill=Object.create(teacher);//以teacher为原型
    bill.job//teacher
    bill.addCourse("soul")//因为Job是原型链上并且没有被继承下来 所以会修改原型链上的值
    //soul.addCourses//soul

window.onload=function(){
        var elem=document.getElementById(“elem”);
        function getData(elem){
            var s=elem.outerHTML.split(“ “)//处理HTML代码以空格分割
            var arr=[];//定义一个arr空数组
            var arrdata=[]//定义一个arrdata数组
            var fg=[]//定义一个fg数组
            var fgname=[]//定义一个fg名称数组
            var obj={}//定义一个返回的obj
            for (var i = 0; i < s.length; i++) {
                //初步处理下data 向arr数组添加data存在的分割数组
                if(s[i].indexOf(“data”)!=-1){
                    arr.push(s[i])
                }
            };
            //处理下数组最后一个带”>”号
            var a=arr[arr.length-1]
            var a_index=a.split(“>”)
            arr[arr.length-1]=a_index[0]
            //处理完毕

            //以=号来分割下数组 所以会生成个二维数组
            for (var i = 0; i < arr.length; i++) {
                var z=arr[i].split(“=”)
                arrdata.push(z)
            };

            //分割二维数组 二维数组0为name 1位data
            for (var i = 0; i < arrdata.length; i++) {
                fg.push(arrdata[i][1])
                fgname.push(arrdata[i][0])
            };
            //数组转化为对象顺便处理下前缀data-
            for (var i = 0; i < fgname.length; i++) {
                var b=fgname[i].replace(/data-/gi,””)
                fgname[i]=b
                obj[fgname[i]]=fg[i]
            };
            console.log(obj)
            return obj;


        }

        getData(elem);
        var m =elem.dataset
        console.log(m)


    }

DOM编程艺术之节点操作 

一:获取节点
        -父子关系
            -parentNode
            -firstChild/lastChild/childNodes
            -childNodes/children

        -collection=element.getElementsByTagName(tagName)//从element获取指定标签名称

        -element.getElementsByTagName("*")//获取element下所有的后代元素
    -getElementsByClassName
        -collection=element.getElementsByClassName(classname)
        -collection可以通过下标来获取指定的元素

    -querySelector/All
        -list=element.querySelector/All(selector)

        -user.querySelectorAll(".user")//获取到user下的class类

    -element.innerText
        -节点及其后代节点内容的文本内容//FF不支持

js判断鼠标进入容器的方向小解

转自http://sentsin.com/web/112.html


$(“#wrap”).bind(“mouseenter mouseleave”,function(e) {
           var w = $(this).width();
           var h = $(this).height();
           var x = (e.pageX - this.offsetLeft - (w / 2))  (w > h ? (h / w) : 1);
           var y = (e.pageY - this.offsetTop - (h / 2))  (h > w ? (w / h) : 1);
           var direction = Math.round((((Math.atan2(y, x) * (180 / Math.PI)) + 180) / 90) + 3) % 4; //direction的值为“0,1,2,3”分别对应着“上，右，下，左”
           var eventType = e.type;
           var dirName = new Array(‘上方’,’右侧’,’下方’,’左侧’);
           if(e.type == ‘mouseenter’){
              $(“#result”).html(dirName[direction]+’进入’);
          }else{
              $(‘#result’).html(dirName[direction]+’离开’);
          }
      });
