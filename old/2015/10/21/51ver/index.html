三:类继承
    -(function(){
        function ClassA(){}
        ClassA.classMethod=function(){}//类对象上 为类方法
        ClassA.prototype.api=function(){}//类原型上方法 为实例上方法

        function ClassB(){
            ClassA.apply(this,arguments)//为了使用构造函数的 new ClassB() 执行下CLASSA
        }

        ClassB.prototype=new ClassA();//给ClassB原型赋值了一个对象
        ClassB.prototype.constructor=ClassB();
        ClassB.prototype.api=function(){//在实例上定义了api
            ClassA.prototype.api.apply(this,arguments)
        }
    })()


    -        ClassA
            calssMethod |function
            prototype     |ClassA.prototype

                ClassA.prototype
                _proto_   |Object.prototype
                api       | function 

        ClassB
        classMethod | function
        prototype     | ClassB.prototype

        ClassB.prototype
        _proto_   |ClassA.prototype
        constructor |ClassA
        api            | function

        B
        _proto_ |ClassB.prototype
