低版本兼容element.children

大概思路就是判断childNodes是否等于1 然后返回一个数组就ok了


window.onload=function(){
        function getElementChildren(elem){
            //如果函数该方法存在
            if(elem.children){
                return elem.children
            }else{}
                var child=[]
                for (var i = 0; i < elem.childNodes.length; i++) {
                    if(elem.childNodes[i].nodeType==1){
                        child.push(elem.childNodes[i])
                    }
                };
                return child;
            }

        }

        var elem=document.getElementById(“elem”)
        var test=getElementChildren(elem)
        console.log(test)
    }
