四:String.prototype.indexOf

-功能

-获取子字符串在字符串中位置索引

-语法

-StringObj.indexOf(searchValue)

var ind=str.indexOf(“he”)//从字符串头部开始查找返回0

str.indexOf(“he”,ind+1)//从上一次位置开始向后查找

str.indexOf(“12131”)//不存在字符串返回-1

var str1=”h e l l o”

str1.split(“ “,3)//[h,e,l]
