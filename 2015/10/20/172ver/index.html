    function tabs (){
        var get_uld=document.getElementById(“uld”)
        var get_ul_li=get_uld.getElementsByTagName(“li”)
        var save_child;
        for (var i = 0; i < get_ul_li.length; i++) {

            get_ul_li[i].onmouseover=function(num){
                return function(){
                    save_child=get_ul_li[num].childNodes[1]
                    showIntor(save_child)
                }
            }(i)
            get_ul_li[i].onmouseout=function(num){
                return function(){
                    save_child=get_ul_li[num].childNodes[1]
                    hide(save_child)
                }
            }(i)
        };



    }
    function hide(elem){

        console.log(elem);
        elem.style.display=”none”
    }
    function showIntor(elem){
        elem.style.display=”block”

    }
    window.onload=function(){
        tabs()
    }
    </script>

    // JS实现选项卡切换
     window.onload = function(){
             var oTab = document.getElementById(“tabs”);
             var oUl = oTab.getElementsByTagName(“ul”)[0];
             var oLis = oUl.getElementsByTagName(“li”);
             var oDivs= oTab.getElementsByTagName(“div”);

             for(var i= 0,len = oLis.length;i<len;i++){
                 oLis[i].index = i;
                 oLis[i].onclick = function() {
                     for(var n= 0;n<len;n++){
                         oLis[n].className = “”;
                         oDivs[n].className = “hide”;
                     }
                     this.className = “on”;
                     oDivs[this.index].className = “”;
                 }
             };
         }

    </script>
