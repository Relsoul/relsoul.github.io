非正则表达式实现低版本elem.dataset

没学过火星文正则表达式~就用最死板的办法来获取了~


window.onload=function(){
        var elem=document.getElementById(“elem”);
        function getData(elem){
            var s=elem.outerHTML.split(“ “)//处理HTML代码以空格分割
            var arr=[];//定义一个arr空数组
            var arrdata=[]//定义一个arrdata数组
            var fg=[]//定义一个fg数组
            var fgname=[]//定义一个fg名称数组
            var obj={}//定义一个返回的obj
            for (var i = 0; i < s.length; i++) {
                //初步处理下data 向arr数组添加data存在的分割数组
                if(s[i].indexOf(“data”)!=-1){
                    arr.push(s[i])
                }
            };
            //处理下数组最后一个带”>”号
            var a=arr[arr.length-1]
            var a_index=a.split(“>”)
            arr[arr.length-1]=a_index[0]
            //处理完毕

            //以=号来分割下数组 所以会生成个二维数组
            for (var i = 0; i < arr.length; i++) {
                var z=arr[i].split(“=”)
                arrdata.push(z)
            };

            //分割二维数组 二维数组0为name 1位data
            for (var i = 0; i < arrdata.length; i++) {
                fg.push(arrdata[i][1])
                fgname.push(arrdata[i][0])
            };
            //数组转化为对象顺便处理下前缀data-
            for (var i = 0; i < fgname.length; i++) {
                var b=fgname[i].replace(/data-/gi,””)
                fgname[i]=b
                obj[fgname[i]]=fg[i]
            };
            console.log(obj)
            return obj;


        }

        getData(elem);
        var m =elem.dataset
        console.log(m)


    }
