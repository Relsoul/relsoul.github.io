JavaScript：延迟切换的选项卡

演示地址:http://www.old.com/demo/CSS-Tab%20%E5%BB%B6%E8%BF%9F.html


 


学习经验


具体的实现方式跟 http://www.old.com/soul-get/soul-js/296


大同小异


唯一不同的是加了个定时器功能


if(timer){
                    clearTimeout(timer)
                    timer=null;
                }
                var that=this;

在开始动画之前判断了 如果定时器存在 那么清除以前的定时器动画 然后把定时器变量设置为null


that=this


这个语句也很重要


首先我们知道 在window对象中 也就是window.setTimeout()通常写成.setTimeout()


在里面使用this永远指向的是window 不信大家可以在定时器中输出下this console.log(this)


然后我们通过在鼠标移动这个动画中建立了变量 把that设置成this 变量在变量域中是存在的


然后的代码就跟上一个实现的方式一样


还有一点不同的是 这次我通过设置fh属性来索引了 大家也可以用soul xxx 之类的来索引


在此就不多谈了

function hasClass(ele,cls) {
    return ele.className.match(new RegExp(‘(\s|^)’+cls+’(\s|$)’));
}

function addClass(ele,cls) {
    if (!this.hasClass(ele,cls)) ele.className += “ “+cls;
}

function removeClass(ele,cls) {
    if (hasClass(ele,cls)) {
        var reg = new RegExp(‘(\s|^)’+cls+’(\s|$)’);
        ele.className=ele.className.replace(reg,’ ‘);
    }
}



    function getClassName(obj,cls){
        var elem=obj.getElementsByTagName(““);
        var ref=[];
        for (var i = 0; i < elem.length; i++) {
            if(elem[i].className==cls){
                ref.push(elem[i]);
            }
        };return ref;
    }

    function changeCss(){
        var get_tab_li=document.getElementById(“title”).getElementsByTagName(“li”)
        var get_tab_con=document.getElementById(“tab-con”).getElementsByTagName(“div”)
        var timer=null;
        //遍历
        if(get_tab_li.length!=get_tab_con.length){return;}
        for (var i = 0; i < get_tab_li.length; i++) {
            get_tab_li[i].setAttribute(“fh”, i)

            get_tab_li[i].onmouseover=function(){
                if(timer){
                    clearTimeout(timer)
                    timer=null;

                var that=this;
            timer=setTimeout(function(){
                for (var j = 0; j < get_tab_li.length; j++) {

                    get_tab_con[j].style.display=”none”
                };
                get_tab_con[that.getAttribute(“fh”)].style.display=”block”;
                addClass(that,”light”)

            }
