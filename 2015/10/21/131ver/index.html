    window.onload=function(){

            var get_img=document.getElementById(“container”).getElementsByTagName(“img”);
            var road_rotate=null,road_top=null,road_left=null;
            for (var i = 0; i < get_img.length; i++) {
                road()
                get_img[i].style.transform=”rotate(“ +road_rotate+”deg)”;
                get_img[i].style.top=road_top+”px”
                get_img[i].style.left=road_left+”px”

                if (i!=0) {
                    var before_left=parseInt(get_img[i-1].style.left);
                    var now_left=parseInt(get_img[i].style.left);
                    console.log(before_left+”|”+now_left);
                    if ((before_left-now_left)<200) {
                        get_img[i].style.left=before_left+200+”px”
                    };

                }


            };

            function road(){
                road_rotate=Math.floor(Math.random()40);
                road_top=Math.floor(Math.random()500);
                road_left=Math.floor(Math.random(4)*800);
                if (road_left<40) {
                    road()
                };
                return {
                    “road_rotate”:road_rotate,
                    “road_top”:road_top,
                    “road_left”:road_left,
                };
            }

    }
